```javascript
~

      { "id": "jem@ionify.net"
      , "is": "a json-expressed module"
      , "by": "mike.lee"
      , "on": -4.200709

      , “at”: +2.1
      , "in":
            [ "forest-hills.new-york.usa.earth"
            ,   "san-jose.california.usa.earth"
            ]
      },

  "do":
      [ "an aesop action"
      , {"if":"do.times", "<":7, "do":"do"}
      ],

  "an aesop action":
      "log Hi! #do.times"
}
;
```
